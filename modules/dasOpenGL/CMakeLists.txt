IF(NOT DAS_OPENGL_INCLUDED)
    SET(DAS_OPENGL_INCLUDED TRUE)
    MESSAGE(STATUS "dasOpenGL module included.")

    # Boost / higher-level layer (pure-das) — shared by desktop and web. These
    # `require opengl` (the raw gl* + GL_ constants layer), which is provided by
    # either the desktop pure-das module (opengl.das, ELSE branch) or the native
    # C++ module (EMSCRIPTEN branch).
    ADD_MODULE_DAS(opengl opengl opengl_boost_internal)
    ADD_MODULE_DAS(opengl opengl opengl_boost)
    ADD_MODULE_DAS(opengl opengl opengl_state)
    ADD_MODULE_DAS(opengl opengl opengl_ttf)
    ADD_MODULE_DAS(opengl opengl opengl_gen)
    ADD_MODULE_DAS(opengl opengl opengl_cache)
    ADD_MODULE_DAS(opengl opengl opengl_imgui_driver)
    ADD_MODULE_DAS(live opengl opengl_live)
    ADD_MODULE_DAS(glsl glsl glsl_opengl)

    # ===== Native C++ GLES3 binding (generated via bind/bind_opengl.das from =====
    # web_include/GLES3/gl3.h). The raw gl* layer bound with exact GLES3 types; wasm
    # call_indirect is strongly typed so the type-erased late-extern trampoline traps,
    # but native makeExterns emit exact-typed glue. SHARED by both branches below:
    #   web (emscripten) -> STATIC module in the wasm monolith; gl* from emcc -sFULL_ES3.
    #   desktop          -> DYNAMIC .shared_module, registered for SIGNATURES only when
    #                       cross-compiling for web (.das_module); gl* are link-only stubs.
    SET(DAS_OPENGL_DIR ${CMAKE_CURRENT_LIST_DIR})
    SET(DAS_OPENGL_BIND_MODULE_SRC
            ${DAS_OPENGL_DIR}/src/need_dasOpenGL.h
            ${DAS_OPENGL_DIR}/src/dasOpenGL.h
            ${DAS_OPENGL_DIR}/src/dasOpenGL.cpp
        # user include (hand-written int64-truncating wrappers for the buffer funcs)
            ${DAS_OPENGL_DIR}/src/dasOpenGL.main.cpp
        # generated binding includes
            ${DAS_OPENGL_DIR}/src/dasOpenGL.alias.add.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.dummy.add.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.enum.add.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.enum.class.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.enum.decl.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.enum.decl.cast.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.struct.add.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.struct.postadd.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.struct.class.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.struct.decl.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.struct.impl.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.struct.cast.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func.reg.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func.decl.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func.aot.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func.aot.decl.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.const.inc
            ${DAS_OPENGL_DIR}/src/dasOpenGL.const.desktop_extras.inc
        # generated functions
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func_1.cpp
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func_2.cpp
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func_3.cpp
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func_4.cpp
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func_5.cpp
            ${DAS_OPENGL_DIR}/src/dasOpenGL.func_6.cpp
        # desktop signatures-only gl* stubs: empty TU under emscripten (real gl* from
        # -sFULL_ES3), link-satisfying no-ops on desktop. Generated by bind_opengl.das.
            ${DAS_OPENGL_DIR}/src/dasOpenGL.desktop_stubs.cpp
        )
    IF(EMSCRIPTEN)
        # Web: STATIC native module, statically pulled into the wasm monolith.
        MESSAGE(STATUS "dasOpenGL: emscripten/web build (native GLES3 C++ binding).")
        ADD_MODULE_CPP(dasOpenGL)
        ADD_DAS_STATIC_MODULE_LIB(libDasModuleOpenGL ${DAS_OPENGL_BIND_MODULE_SRC})
        TARGET_INCLUDE_DIRECTORIES(libDasModuleOpenGL PRIVATE ${DAS_OPENGL_DIR}/web_include)
        SETUP_CPP11(libDasModuleOpenGL)
    ELSE()
        # Desktop: pure-das late-extern opengl.das stays the runtime module.
        ADD_MODULE_DAS(opengl opengl opengl)
        # PLUS the native module as a DYNAMIC .shared_module ONLY -- deliberately NOT
        # ADD_MODULE_CPP (that PULL_MODULEs it statically and would double-register
        # "opengl" against opengl.das). Loaded on demand by .das_module's
        # register_dynamic_module when this host cross-compiles for web, to supply
        # Module("opengl") signatures (the real GLES3 glue links into the wasm target).
        # Skipped on win32: that lane forces DAS_LLVM_DISABLED (no wasm cross-compile,
        # so the signatures-only native module is purposeless), and the vendored GLES3
        # GL_APIENTRY (__stdcall on Win32) doesn't match the makeExtern glue.
        IF(WIN32 AND CMAKE_SIZEOF_VOID_P EQUAL 4)
            MESSAGE(STATUS "dasOpenGL: skipping native dasModuleOpenGL on win32 (no wasm cross-compile).")
        ELSE()
            MESSAGE(STATUS "dasOpenGL: desktop build (opengl.das runtime + signatures-only dasModuleOpenGL.shared_module).")
            ADD_DAS_SHARED_MODULE_LIB(dasModuleOpenGL ${DAS_OPENGL_BIND_MODULE_SRC})
            TARGET_INCLUDE_DIRECTORIES(dasModuleOpenGL PRIVATE ${DAS_OPENGL_DIR}/web_include)
            SETUP_CPP11(dasModuleOpenGL)
        ENDIF()
    ENDIF()

    install(DIRECTORY ${PROJECT_SOURCE_DIR}/modules/dasOpenGL/opengl
        DESTINATION ${DAS_INSTALL_MODULESDIR}/dasOpenGL
        FILES_MATCHING
        PATTERN "*.das"
        PATTERN "_aot_generated" EXCLUDE
    )

    install(DIRECTORY ${PROJECT_SOURCE_DIR}/modules/dasOpenGL/glsl
        DESTINATION ${DAS_INSTALL_MODULESDIR}/dasOpenGL
        FILES_MATCHING
        PATTERN "*.das"
        PATTERN "_aot_generated" EXCLUDE
    )

ENDIF()
